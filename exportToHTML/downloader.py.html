69       def get_date_list(self, from_date, to_date):
70           from_date = datetime.datetime.strptime(from_date, '%Y/%m/%d')
71           to_date = datetime.datetime.strptime(to_date, '%Y/%m/%d')
72           diff = (to_date - from_date).days
73           if diff <= 0:
74               messagebox.showwarning(title='警告', message='日期范围有误')
75           else:
76               date_list = [from_date.strftime('%Y/%m/%d')]
77               for i in range(1, diff+1):
78                   date = from_date + datetime.timedelta(days=i)
79                   date = date.strftime('%Y/%m/%d')
80                   date_list.append(date)
81               return date_list
