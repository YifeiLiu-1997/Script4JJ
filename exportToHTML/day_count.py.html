5    def g_time(chapter, how_many_times_one_chapter):
6        plan = {}
7        start_time = datetime.datetime.strptime('2022-06-07', '%Y-%m-%d')
8        start_plan = 0
9        weeks = (chapter * how_many_times_one_chapter) // 7
10       for i in range(weeks):
11           start_time += datetime.timedelta(days=7)
12           start_plan += 7 / how_many_times_one_chapter
13           if start_plan > chapter:
14               start_plan = chapter
15           plan[start_time.strftime('%Y-%m-%d')] = start_plan

20   def day_after(today, days):
21       return (datetime.datetime.strptime(today, '%Y-%m-%d') + datetime.timedelta(days=days)).strftime('%Y-%m-%d')
